دليل تثبيت Manim CE على Windows

# 1. فتح موجه الأوامر (CMD) أو PowerShell
# 2. إنشاء مجلد للمشروع
mkdir manim-project
cd manim-project

# 3. إنشاء بيئة افتراضية
python -m venv venv

# 4. تفعيل البيئة الافتراضية
venv\Scripts\activate

# 5. ترقية pip
pip install --upgrade pip

# 6. تثبيت Manim
pip install manim

# 7. تثبيت الحزم الإضافية
pip install numpy scipy matplotlib ipython

# 8. (اختياري) تثبيت إصدار التطوير
pip install "manim[dev]"

# 9. التحقق من التثبيت
manim --version

# 10. اختبار التثبيت بمثال بسيط
manim -pql example.py SquareToCircle
